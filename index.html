   import java.util.*;


public class Divider

  {

  private double Vtop;

  private double Vbottom;

  private double Rtop;

  private double Rbottom;

  private double Vout;

  private double Itotal;

  

  public Divider(double vtop, double vbottom, double rtop, double rbottom)

  {

  Vtop = vtop;

  Vbottom = vbottom;

  Rtop = rtop;

  Rbottom = rbottom;

  }

  public double setVtop(double vtop)

  {

  return Vtop = vtop;

  }

  public double setVbottom(double vbottom)

  {

  return Vbottom = vbottom;

  }

  

  public double setRtop(double rtop) 

  {

  return Rtop = rtop;

  }

  

  public double setRbottom(double rbottom)

  {

  return Rbottom = rbottom;

  }

  public double getVout()

  {

  return Vout;

  }

  public double getItotal()

  {

  return Itotal;

  }


 public void Calculate()

  {

  Vout = (Rtop/(Rtop+Rbottom))*(Vtop-Vbottom);

  Itotal=(Vtop-Vbottom)/(Rtop+Rbottom);

  }

  public void display()

  {

  

  //NumberFormat df = new NumberFormat.getInstance();

  

  System.out.println("Vout ="+Vout+"V");

  System.out.println("Current ="+Itotal*1000+"mA");

  }

  

  public static void main(String[] args)

  {

  Divider jerome= new Divider(10, -10, 5000, 15000);
